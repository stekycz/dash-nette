480:      * @param  int

483:     public static function replace($subject, $pattern, $replacement = NULL, $limit = -1)
484:     {
485:         if (is_object($replacement) || is_array($replacement)) {
486:             if ($replacement instanceof Nette\Callback) {

506:     {
507:         $res = Callback::invokeSafe($func, $args, function($message) use ($args) {
508:             // compile-time error, not detectable by preg_last_error
