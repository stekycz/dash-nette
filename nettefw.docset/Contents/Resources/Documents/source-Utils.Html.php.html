496:      * Returns element's attributes.
497:      * @return string
498:      */
499:     public function attributes()
500:     {
501:         if (!is_array($this->attrs)) {
502:             return '';
503:         }
504: 
505:         $s = '';
506:         foreach ($this->attrs as $key => $value) {
507:             if ($value === NULL || $value === FALSE) {
508:                 continue;
509: 
510:             } elseif ($value === TRUE) {
511:                 if (static::$xhtml) {
512:                     $s .= ' ' . $key . '="' . $key . '"';
513:                 } else {
514:                     $s .= ' ' . $key;
515:                 }
516:                 continue;
517: 
518:             } elseif (is_array($value)) {
519:                 if ($key === 'data') { // deprecated
520:                     foreach ($value as $k => $v) {
521:                         if ($v !== NULL && $v !== FALSE) {
522:                             if (is_array($v)) {
523:                                 $v = Json::encode($v);
524:                             }
525:                             $q = strpos($v, '"') === FALSE ? '"' : "'";
526:                             $s .= ' data-' . $k . '='
527:                                 . $q . str_replace(array('&', $q), array('&amp;', $q === '"' ? '&quot;' : '&#39;'), $v)
528:                                 . (strpos($v, '`') !== FALSE && strpbrk($v, ' <>"\'') === FALSE ? ' ' : '')
529:                                 . $q;
530:                         }
531:                     }
532:                     continue;
533: 
534:                 } elseif (strncmp($key, 'data-', 5) === 0) {
535:                     $value = Json::encode($value);
536: 
537:                 } else {
538:                     $tmp = NULL;
539:                     foreach ($value as $k => $v) {
540:                         if ($v != NULL) { // intentionally ==, skip NULLs & empty string
541:                             //  composite 'style' vs. 'others'
