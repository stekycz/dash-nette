150:     public function move($dest)
151:     {
152:         @mkdir(dirname($dest), 0777, TRUE); // @ - dir may already exist

154:         if (!call_user_func(is_uploaded_file($this->tmpName) ? 'move_uploaded_file' : 'rename', $this->tmpName, $dest)) {
